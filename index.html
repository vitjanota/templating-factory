Example 3

Template with repetitive data inside (usage of for-each directive) and two data entries to process

Data

{
  item01: {
    title: "Fruits",
    text: "Among fruits belong",
    color: "red",
    list: [
      {
        content:"apple",
        color: "#e0c900"
      },
      {
        content:"orange",
        color: "orange"
      }
    ]
  },
  item02: {
    title: "Vegetables",
    text: "Among vegetables belong",
    color: "green",
    list: [
      {
        content:"carrot",
        color: "orange"
      },
      {
        content:"tomato",
        color: "red"
      },
      {
        content:"cucumber",
        color: "green"
      }
    ]
  }
}

Template(s)

<div style="border: 2px solid %%color%%">
  <h1>%%title%%</h1>
  <p>%%text%%:</p>
  <ul>
    <li data-for-each="list">
      <span style="color: %%color%%">%%content%%</span>
    </li>
  </ul>
  <p>and many more.</p>
</div>

Example 5

Simple setup for rich text formatting (usage of process directive)

Data

{
  par: {
    color: "blue",
    content: [
      {text: "This example
        shows how text with
        inner formatting like "},
      {text: "bold",
        type: "emph"},
      {text: ", "},
      {text: "italic",
        type: "slant"},
      {text: ", "},
      {text: "link",
        url: "#",
        type: "link"},
      {text: " and others
        can be processed."}
      {type: 'lbr'},
      {type: 'lbr'},
      {text: "And don't forget: "},
      {text: "This is data driven!",
        type: "emph"}
  }
}

Template(s)

<div style="border:2px solid %%color%%"
     data-process="content">
  <span>%%text%%</span>
  <b data-type="emph">%%text%%</b>
  <i data-type="slant">%%text%%</i>
  <a href="%%url%%" data-type="link">%%text%%</a>
  <br data-type='lbr'/>
</div>
%%text%% %%text%% %%text%% %%text%%

Example 6

Simple table processing

Data

{
    table: {
      header: [
        {
          cells: [
            {content: 'header 1'},
            {content: 'header 2'}
          ]
        }
      ],
      body: [
        {
          cells: [
            {content: 'cell 1.1'},
            {content: 'cell 1.2'}
          ]
        },
        {
          cells: [
            {content: 'cell 2.1'},
            {content: 'cell 2.2'}
          ]
        }
      ],
      footer: [
        {
          cells: [
            {content: 'footer 1'},
            {content: 'footer 2'}
          ]
        }
      ]
    }
  }

Template(s)

<table>
  <thead data-for-each="header">
    <tr data-process="cells">
      <th>%%content%%</th>
    </tr>
  </thead>
  <tbody data-process="body">
    <tr data-process="cells">
      <td>%%content%%</td>
    </tr>
  </tbody>
  <tfoot data-for-each="footer">
    <tr data-process="cells">
      <td>%%content%%</td>
    </tr>
  </tfoot>
</table>
%%content%%
